2. **Async Excel Parser (Non-blocking)**
   - File: `backend/app/services/excel_parser.py`
     ```python
     import asyncio
     import pandas as pd
     from fastapi import UploadFile, HTTPException
     from app.config import settings

     async def parse_excel(file: UploadFile):
         if not file.filename.lower().endswith((".xlsx", ".xls")):
             raise HTTPException(422, detail={"loc": ["body", "file"], "msg": "Invalid extension"})
         content = await file.read()
         if len(content) > settings.MAX_UPLOAD_BYTES:
             raise HTTPException(413, detail={"loc": ["body", "file"], "msg": "File too large"})
         df = await asyncio.to_thread(pd.read_excel, content)
         return {"rows": df.fillna("").to_dict(orient="records")}
     ```

5. **API Routes**
   - File: `backend/app/api/uploads.py`
     ```python
     from fastapi import APIRouter, UploadFile, File, HTTPException
     from fastapi.responses import FileResponse
     from app.services.excel_parser import parse_excel

     router = APIRouter()

     @router.post("/upload")
     async def upload(file: UploadFile = File(...)):
         return await parse_excel(file)

     @router.get("/download/{template}")
     async def download_template(template: str):
         allowed = {"single_work_nit.xlsx", "multi_work_nit.xlsx"}
         if template not in allowed:
             raise HTTPException(422, detail="Invalid template")
         return FileResponse(f"templates/downloads/{template}", filename=template)

     @router.get("/health")
     async def health():
         return {"status": "ok"}
     ```

6. **Rate Limiter**

     @app.on_event("startup")

## 📅 Week-04: Authentication & Finalization – Both Teams

### Backend (In-House Team)
1. **User Authentication**
   - File: `backend/app/main.py` (add to existing file)
     ```python
     from fastapi.security import OAuth2PasswordBearer, OAuth2PasswordRequestForm
     from fastapi import Depends, HTTPException, status
     from jose import jwt, JWTError
     from app.config import settings

     oauth2_scheme = OAuth2PasswordBearer(tokenUrl="token")

     async def get_current_user(token: str = Depends(oauth2_scheme)):
         try:
             payload = jwt.decode(token, settings.SECRET_KEY, algorithms=[settings.ALGORITHM])
             username: str = payload.get("sub")
             if username is None:
                 raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid credentials")
             return username
         except JWTError:
             raise HTTPException(status_code=status.HTTP_401_UNAUTHORIZED, detail="Invalid credentials")
     ```

   - File: `tender-frontend/src/App.tsx` (update existing file)
     ```typescript
     import { useAuth0 } from '@auth0/auth0-react';
     import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom';
     import { QueryClient, QueryClientProvider } from '@tanstack/react-query';
     import { Toaster } from 'react-hot-toast';
     import UploadPage from './pages/UploadPage';
     import DownloadPage from './pages/DownloadPage';

     const queryClient = new QueryClient();

     const PrivateRoute = ({ children }: { children: JSX.Element }) => {
       const { isAuthenticated } = useAuth0();
       return isAuthenticated ? children : <Navigate to="/" />;
     };

     export default function App() {
       const { isLoading } = useAuth0();
       if (isLoading) return <div>Loading...</div>;
       return (
         <QueryClientProvider client={queryClient}>
           <BrowserRouter>
             <Routes>
               <Route path="/" element={<UploadPage />} />
               <Route path="/download" element={<PrivateRoute><DownloadPage /></PrivateRoute>} />
             </Routes>
             <Toaster position="top-center" reverseOrder={false} />
           </BrowserRouter>
         </QueryClientProvider>
       );
     }
     ```
   - Wrap `App` in `index.tsx`:
     ```typescript

     import App from './App';

       <Auth0Provider
         domain="your-domain.auth0.com"
         clientId="your-client-id"
         redirectUri={window.location.origin}
       >
         <App />
       </Auth0Provider>,
       document.getElementById('root')
     );
     ```
